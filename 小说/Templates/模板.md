<%*
// 1. 获取当前活动文件
const file = tp.config.active_file;
// 2. 读取文件现有内容（保留你下载的 Gemini 对话）
let content = await app.vault.read(file);

// 3. 检查是否已经有属性头，避免重复添加
if (!content.startsWith("---")) {
    const time = tp.date.now()
    
    // 4. 定义你的 Gemini 属性模板
    const yamlHeader = `---
生成时间: ${time}
状态: 原始/待筛选/潜力股/已使用
标签: 
---

`;
    // 5. 核心：强制将属性区和原内容合并，写回文件
    await app.vault.modify(file, yamlHeader + content);
}
%>
